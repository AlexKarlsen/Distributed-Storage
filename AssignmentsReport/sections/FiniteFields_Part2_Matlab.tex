\section{Finite Fields - Part 2}
\begin{par}
    \begin{flushleft}
    We are to construct the addition and multiplication table for $\mathrm{GF}\left(2^3 \right)$ with the primitive polynomail $x^3 +x^2 +1$
    \end{flushleft}
    \end{par}
    
    \begin{matlabcode}
    clear;clc;
    x = [];
    for i = 0:7
        for j = 0:7
            x = [x ; [i j]];
        end
    end
    gf = gf(x,3,'D3+D2+1');
    
    % Addition
    AddTable = zeros(7,7,'uint32');
    for i = 1:8
        for j = 1:8
            z = j +  (i-1)*8;
            v1 = gf(z,1);
            v2 = gf(z,2);
            tmp = v1 + v2;
            AddTable(i,j) = tmp.x;
        end
    end
    
    % Multiply
    MulTable = zeros(7,7,'uint32');
    for i = 1:8
        for j = 1:8
            z = j +  (i-1)*8;
            v1 = gf(z,1);
            v2 = gf(z,2);
            tmp = v1 * v2;
            MulTable(i,j) = tmp.x;
        end
    end
    
    \end{matlabcode}
